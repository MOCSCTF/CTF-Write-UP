```python
def clock(pix):
  switcher={
    (255, 204, 0):1,

    (255, 51, 0):3,	
    (254, 0, 0):4 ,	
    (152, 0, 101):5,

    (0, 153, 0):'a',
    (0, 255, 1):'b',
    (255, 255, 0):0,
	(255, 255, 255):' '
    }
  return (switcher.get(pix,"Invalid arg"))
```
